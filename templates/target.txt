#Configuration for mymake
#Lines beginning with # are comments.
#Any unknown left-hand sides can be used as variables.

[]

#Extensions for source-files, ie. files that should be compiled.

ext+=cc
ext+=cxx
ext+=c++
ext+=c

#Build directory

#Executable directory
execDir=bin/

#Ignore files.
ignore=*.template.*

#input=Xxx.cpp

#Output filename, defaults to first input file.
#output=

#Include path

#library+=m

#Include local libraries (eg. in other projects). Usually done automatically.
#localLibrary+=mylib

#Run output by default

#Show compilation time?
#showTime=no

#Precompiled header.
#pch=std.h

#Pre-build steps.
#preBuild=

#Post-build steps.
#postBuild=

#Defines.
#define+=DEBUG


#Define command line. Should not need to be changed.
defines=<defineCl*define>

#Static library
[lib]

#Compile all files into a lib.
input=*

#Do not run.
execute=no

#Link the output of this target into any dependent targets.
linkOutput=yes

#Forward dependencies.
forwardDeps=yes

#Shared library
[sharedlib]

#Compile all files.
input=*

#Do not run.
execute=no


[release]
#Output somewhere else.
buildDir=release/build/
execDir=release/bin/

[unix]

execExt=
intermediateExt=o

includeCl=-iquote
libraryCl=-l
localLibraryCl=
defineCl=-D 

flags=-std=c++11 -Wall -Wextra

pchFile=stdafx.h.gch
usePch=-include <pchFile>

#Compile the precompiled header.
pchFile=stdafx.h.gch
pchCompile=g++ <flags> <file> -c <includes> -o <output>
pchCompileCombined=no
usePch=-include <noExt|pchFile>

#Compile a single file.
compile=*:g++ <flags> <usePch*if|pch> <file> -c <includes> -o <output>

#Link all files
link=g++ <files> <libs> -o <output>

[unix,!release]
flags+=-g

[unix,release]
flags+=-O3

[unix,lib]

execExt=a

#Link (more flags for libraries?)
link=ar rcs <output> <files>

[unix,sharedlib]
execExt=so
link=g++ -shared <files> <libs> -o <output>

[windows]

execExt=exe
intermediateExt=obj

includeCl=/I
libraryCl=
localLibraryCl=
defineCl=/D

flags=/nologo /EHsc /W3
linkFlags=/MANIFEST

pchFile=stdafx.pch
usePch=/Yu
namePch=/Fp<pchFile>

pchCompile=cl <defines> <file> <includes> <flags> /Yc<title|pch> <namePch*if|pch> /c /Fo<output>
pchCompileCombined=yes

compile=*:cl <defines> <file> <includes> <flags> <usePch*title|pch> <namePch*if|pch> /c /Fo<output>
link=link <linkFlags> <libs> <files> /nologo /OUT:<output>
